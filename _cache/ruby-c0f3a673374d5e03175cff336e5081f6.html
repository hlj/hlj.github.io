def atomize(site, type, posts)
  path = "/#{type}/#{posts[0]}"
  atom = AtomPage.new(site, site.source, path, type, posts[0])
  site.pages << atom
end

class AtomPage < Page
  def initialize(site, base, dir, type, val)
    @site = site
    @base = base
    @dir = dir
    @name = 'atom.xml'

    self.process(@name)
    self.read_yaml(File.join(base, '_layouts'), "#{type}_atom.xml")
    self.data[type] = val
  end
end
