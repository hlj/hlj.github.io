require 'java'

class java::util::concurrent::locks::ReentrantReadWriteLock
    def read(&block)
        read_lock = self.read_lock
        read_lock.lock
        begin
            block.call
        ensure
            read_lock.unlock
        end
    end

    def write(&block)
        write_lock = self.write_lock
        write_lock.lock
        begin
            block.call
        ensure
            write_lock.unlock
        end
    end
end
